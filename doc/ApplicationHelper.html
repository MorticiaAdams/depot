def hidden_div_if(condition, attributes = (), &block)
    if condition
        attributes["style"] = "display: none"
    end
    content_tag("div", attributes, &block)
end

def is_admin?(user)
    admin_role = ROLES.find(:first, :conditions => ["name = ?", "admin"])
    return user.roles.include?(admin_role)
end
